**WorkshopSpatialAdapter** (`workshop_spatial_adapter.gd`):
```gdscript
class_name WorkshopSpatialAdapter
extends Node2D

# Renders FloorLayout domain object to Godot nodes
# Creates child Sprite2D + StaticBody2D for each equipment entity

signal equipment_rendered(equipment_id: String)

var _floor_layout: FloorLayout = null
var _equipment_nodes: Dictionary = {}  # equipment_id -> Node2D

# Y-sort group for equipment sprites
@onready var _equipment_container: Node2D = $EquipmentContainer

func setup(layout: FloorLayout) -> void:
    _floor_layout = layout
    _clear_equipment()
    _render_equipment()

func _clear_equipment() -> void:
    for node in _equipment_nodes.values():
        node.queue_free()
    _equipment_nodes.clear()

func _render_equipment() -> void:
    if _floor_layout == null:
        return
    
    for equipment in _floor_layout.equipment_slots:
        var node = _create_equipment_node(equipment)
        _equipment_container.add_child(node)
        _equipment_nodes[equipment.equipment_id] = node
        equipment_rendered.emit(equipment.equipment_id)

func _create_equipment_node(equipment: EquipmentEntity) -> Node2D:
    var root = Node2D.new()
    root.name = equipment.equipment_id
    
    # Position in isometric world coordinates
    var world_pos = _tile_to_world(equipment.tile_position)
    root.position = world_pos
    
    # Sprite
    var sprite = Sprite2D.new()
    sprite.name = "Sprite"
    if equipment.sprite_path and ResourceLoader.exists(equipment.sprite_path):
        sprite.texture = load(equipment.sprite_path)
    else:
        # Use placeholder colored rect
        sprite.texture = _create_placeholder_texture(equipment.equipment_type)
    root.add_child(sprite)
    
    # Collision body for player blocking
    var body = StaticBody2D.new()
    body.name = "CollisionBody"
    var collision = CollisionShape2D.new()
    var shape = RectangleShape2D.new()
    shape.size = equipment.collision_rect.size
    collision.shape = shape
    # Center collision on sprite
    collision.position = Vector2(0, equipment.collision_rect.size.y / 4)
    body.add_child(collision)
    root.add_child(body)
    
    return root

func _tile_to_world(tile_pos: Vector2i) -> Vector2:
    # Convert isometric tile coordinates to world position
    var world_x = (tile_pos.x - tile_pos.y) * 32  # Half tile width (64/2)
    var world_y = (tile_pos.x + tile_pos.y) * 16  # Half tile height (32/2)
    return Vector2(world_x, world_y)

func _create_placeholder_texture(type: EquipmentEntity.EquipmentType) -> Texture2D:
    # Create simple colored placeholder texture
    var image = Image.create(64, 48, false, Image.FORMAT_RGBA8)
    var color: Color
    match type:
        EquipmentEntity.EquipmentType.WORKBENCH:
            color = Color(0.6, 0.4, 0.2)  # Brown
        EquipmentEntity.EquipmentType.LOCKER:
            color = Color(0.5, 0.5, 0.6)  # Gray
        EquipmentEntity.EquipmentType.CRATE:
            color = Color(0.7, 0.5, 0.3)  # Tan
        EquipmentEntity.EquipmentType.SHELVING:
            color = Color(0.4, 0.3, 0.2)  # Dark brown
    image.fill(color)
    return ImageTexture.create_from_image(image)

func get_equipment_node(equipment_id: String) -> Node2D:
    return _equipment_nodes.get(equipment_id)
```

**WorkshopLayoutController** (`workshop_layout_controller.gd`):
```gdscript
class_name WorkshopLayoutController
extends Node2D

# Orchestrates domain loading and adapter rendering for Workshop car
# Attach this script to the Workshop scene root

@export var layout_resource_path: String = "res://src/train/cars/workshop/data/workshop_floor_layout.tres"

var _floor_layout: FloorLayout = null
var _collision_grid: FloorCollisionGrid = null

@onready var _spatial_adapter: WorkshopSpatialAdapter = $WorkshopSpatialAdapter
@onready var _floor_tilemap: TileMapLayer = $FloorTileMap

signal workshop_ready

func _ready() -> void:
    _load_layout()
    _setup_floor_tiles()
    _render_equipment()
    workshop_ready.emit()

func _load_layout() -> void:
    _floor_layout = FloorLayoutLoader.load_from_resource(layout_resource_path)
    if _floor_layout == null:
        push_error("WorkshopLayoutController: Failed to load layout from " + layout_resource_path)
        return
    
    _collision_grid = FloorCollisionGrid.new(_floor_layout)

func _setup_floor_tiles() -> void:
    if _floor_layout == null or _floor_tilemap == null:
        return
    
    # Clear existing tiles
    _floor_tilemap.clear()
    
    # Fill floor area with tiles
    # Using tile ID 0 as floor tile (configure in TileSet)
    for x in range(_floor_layout.width_tiles):
        for y in range(_floor_layout.height_tiles):
            _floor_tilemap.set_cell(Vector2i(x, y), 0, Vector2i(0, 0))

func _render_equipment() -> void:
    if _floor_layout == null or _spatial_adapter == null:
        return
    
    _spatial_adapter.setup(_floor_layout)

func is_position_blocked(world_pos: Vector2) -> bool:
    if _collision_grid == null:
        return false
    return _collision_grid.is_blocked(world_pos)

func get_equipment_at(world_pos: Vector2) -> EquipmentEntity:
    if _floor_layout == null:
        return null
    return _floor_layout.get_equipment_at(world_pos)
```
