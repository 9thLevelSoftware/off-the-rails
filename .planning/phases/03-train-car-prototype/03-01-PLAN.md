**FloorLayout** (`floor_layout.gd`):
```gdscript
class_name FloorLayout
extends RefCounted

# Floor dimensions in isometric tile units (64x32 per tile)
var width_tiles: int = 0
var height_tiles: int = 0

# Equipment slots - list of EquipmentEntity
var equipment_slots: Array[EquipmentEntity] = []

# Floor boundaries for collision
var floor_bounds: Rect2 = Rect2()

func _init(w: int = 0, h: int = 0) -> void:
    width_tiles = w
    height_tiles = h
    floor_bounds = Rect2(0, 0, w * 64, h * 32)

func add_equipment(entity: EquipmentEntity) -> void:
    equipment_slots.append(entity)

func get_equipment_at(iso_pos: Vector2) -> EquipmentEntity:
    # Return equipment entity at given isometric position, or null
    for eq in equipment_slots:
        if eq.collision_rect.has_point(iso_pos):
            return eq
    return null

func is_position_valid(iso_pos: Vector2) -> bool:
    # Check if position is within floor bounds and not blocked
    return floor_bounds.has_point(iso_pos)
```

```gdscript
var state: EquipmentState = EquipmentState.OPERATIONAL

# Position in isometric tile coordinates
var tile_position: Vector2i = Vector2i.ZERO

# Collision rect in isometric world coordinates
var collision_rect: Rect2 = Rect2()

# Sprite resource path for adapter to load

    equipment_id = id
    equipment_type = type

    tile_position = tile_pos
    # Convert tile position to isometric world position
    var world_x = (tile_pos.x - tile_pos.y) * 32  # Half tile width
    var world_y = (tile_pos.x + tile_pos.y) * 16  # Half tile height
    collision_rect = Rect2(Vector2(world_x, world_y), collision_size)
```

Create a spatial query system for efficient collision detection. Use a simple grid-based approach for O(1) lookups.

**FloorCollisionGrid** (`floor_collision_grid.gd`):
```gdscript
class_name FloorCollisionGrid
extends RefCounted

# Grid cell size in world units (one isometric tile)
const CELL_SIZE: Vector2 = Vector2(64, 32)

# Grid storage: cell_key -> Array of EquipmentEntity
var _grid: Dictionary = {}

# Reference to the floor layout
var _layout: FloorLayout = null

func _init(layout: FloorLayout = null) -> void:
    _layout = layout
    if layout:
        _build_grid()

func _build_grid() -> void:
    _grid.clear()
    for equipment in _layout.equipment_slots:
        _add_to_grid(equipment)

func _cell_key(world_pos: Vector2) -> String:
    var cell_x = int(world_pos.x / CELL_SIZE.x)
    var cell_y = int(world_pos.y / CELL_SIZE.y)
    return "%d,%d" % [cell_x, cell_y]

func _add_to_grid(equipment: EquipmentEntity) -> void:
    # Add equipment to all cells its collision rect overlaps
    var rect = equipment.collision_rect
    var min_cell = Vector2(rect.position.x / CELL_SIZE.x, rect.position.y / CELL_SIZE.y).floor()
    var max_cell = Vector2(rect.end.x / CELL_SIZE.x, rect.end.y / CELL_SIZE.y).ceil()
    
    for x in range(int(min_cell.x), int(max_cell.x) + 1):
        for y in range(int(min_cell.y), int(max_cell.y) + 1):
            var key = "%d,%d" % [x, y]
            if not _grid.has(key):
                _grid[key] = []
            _grid[key].append(equipment)

func query_at(world_pos: Vector2) -> Array[EquipmentEntity]:
    # Return all equipment entities whose collision rect contains the position
    var key = _cell_key(world_pos)
    if not _grid.has(key):
        return []
    
    var result: Array[EquipmentEntity] = []
    for equipment in _grid[key]:
        if equipment.collision_rect.has_point(world_pos):
            result.append(equipment)
    return result

func is_blocked(world_pos: Vector2) -> bool:
    return not query_at(world_pos).is_empty()

func query_rect(rect: Rect2) -> Array[EquipmentEntity]:
    # Return all equipment that intersects with the given rect
    var result: Array[EquipmentEntity] = []
    var checked: Dictionary = {}  # Avoid duplicates
    
    var min_cell = Vector2(rect.position.x / CELL_SIZE.x, rect.position.y / CELL_SIZE.y).floor()
    var max_cell = Vector2(rect.end.x / CELL_SIZE.x, rect.end.y / CELL_SIZE.y).ceil()
    
    for x in range(int(min_cell.x), int(max_cell.x) + 1):
        for y in range(int(min_cell.y), int(max_cell.y) + 1):
            var key = "%d,%d" % [x, y]
            if _grid.has(key):
                for equipment in _grid[key]:
                    if not checked.has(equipment.equipment_id):
                        if equipment.collision_rect.intersects(rect):
                            result.append(equipment)
                            checked[equipment.equipment_id] = true
    return result
```

**FloorLayoutLoader** (`floor_layout_loader.gd`):
```gdscript
class_name FloorLayoutLoader
extends RefCounted

# Load a FloorLayout from a .tres resource file
# Resource format expected:
# {
#   "width_tiles": int,
#   "height_tiles": int,
#   "equipment": [
#     { "id": "workbench_1", "type": "WORKBENCH", "tile_x": 2, "tile_y": 1, "collision_width": 64, "collision_height": 48, "sprite": "res://..." },
#     ...
#   ]
# }
static func load_from_resource(resource_path: String) -> FloorLayout:
    if not ResourceLoader.exists(resource_path):
        push_error("FloorLayoutLoader: Resource not found: " + resource_path)
        return null
    
    var resource = ResourceLoader.load(resource_path)
    if resource == null:
        push_error("FloorLayoutLoader: Failed to load resource: " + resource_path)
        return null
    
    return _parse_resource(resource)

static func _parse_resource(resource: Resource) -> FloorLayout:
    # Handle both Dictionary resources and custom Resource classes
    var data: Dictionary = {}
    
    if resource is Resource:
        # Try to get data from exported properties or as dictionary
        if resource.has_method("get_data"):
            data = resource.get_data()
        elif "data" in resource:
            data = resource.data
        else:
            # Assume the resource properties map directly
            data = {
                "width_tiles": resource.get("width_tiles") if "width_tiles" in resource else 5,
                "height_tiles": resource.get("height_tiles") if "height_tiles" in resource else 6,
                "equipment": resource.get("equipment") if "equipment" in resource else []
            }
    
    return _build_layout(data)

static func _build_layout(data: Dictionary) -> FloorLayout:
    var layout = FloorLayout.new(
        data.get("width_tiles", 5),
        data.get("height_tiles", 6)
    )
    
    var equipment_data: Array = data.get("equipment", [])
    for eq_dict in equipment_data:
        var entity = _build_equipment(eq_dict)
        if entity:
            layout.add_equipment(entity)
    
    return layout

static func _build_equipment(eq_dict: Dictionary) -> EquipmentEntity:
    var id: String = eq_dict.get("id", "unknown")
    var type_str: String = eq_dict.get("type", "WORKBENCH")
    var type: EquipmentEntity.EquipmentType = _parse_equipment_type(type_str)
    
    var entity = EquipmentEntity.new(id, type)
    
    var tile_pos = Vector2i(
        eq_dict.get("tile_x", 0),
        eq_dict.get("tile_y", 0)
    )
    var collision_size = Vector2(
        eq_dict.get("collision_width", 64),
        eq_dict.get("collision_height", 32)
    )
    
    entity.set_position(tile_pos, collision_size)
    entity.sprite_path = eq_dict.get("sprite", "")
    
    return entity

static func _parse_equipment_type(type_str: String) -> EquipmentEntity.EquipmentType:
    match type_str.to_upper():
        "WORKBENCH": return EquipmentEntity.EquipmentType.WORKBENCH
        "LOCKER": return EquipmentEntity.EquipmentType.LOCKER
        "CRATE": return EquipmentEntity.EquipmentType.CRATE
        "SHELVING": return EquipmentEntity.EquipmentType.SHELVING
        _: return EquipmentEntity.EquipmentType.WORKBENCH
```
