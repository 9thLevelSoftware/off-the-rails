## Task 1: Create Passive Bonus Manager

**Action:**
1. Create `src/professions/passive_bonus_manager.gd`:
   ```gdscript
   class_name PassiveBonusManager
   extends Node
   
   ## Manages passive bonus application from profession data.
   ## Uses PassiveBonusMapping from Plan 05-01 for string → struct conversion.
   ## Other systems call apply_modifier() to get bonus-adjusted values.
   
   signal bonuses_changed
   
   var _profession: ProfessionData = null
   var _active_bonuses: Array[Dictionary] = []  # [{stat, value, type}, ...]
   
   func set_profession(profession: ProfessionData) -> void:
       _profession = profession
       _active_bonuses.clear()
       
       if profession == null:
           bonuses_changed.emit()
           return
       
       # Use PassiveBonusMapping to convert string bonuses to structured data
       _active_bonuses = PassiveBonusMapping.get_bonuses_for_profession(profession)
       
       print("[PassiveBonusManager] Loaded %d bonuses for %s" % [_active_bonuses.size(), profession.name])
       for bonus in _active_bonuses:
           print("  - %s: %.2f (%s)" % [bonus.stat, bonus.value, bonus.type])
       
       bonuses_changed.emit()
   
   ## Apply modifier to a base value for a given stat.
   ## Returns modified value, or base if no bonus exists.
   func apply_modifier(stat: String, base_value: float) -> float:
       for bonus in _active_bonuses:
           if bonus.stat == stat:
               match bonus.type:
                   "multiply":
                       return base_value * bonus.value
                   "add":
                       return base_value + bonus.value
                   "flag":
                       # Flags don't modify numeric values
                       return base_value
       return base_value
   
   ## Check if a specific bonus is active.
   func has_bonus(stat: String) -> bool:
       for bonus in _active_bonuses:
           if bonus.stat == stat:
               return true
       return false
   
   ## Check if a flag bonus is active (for non-numeric bonuses like "detect hidden").
   func has_flag(stat: String) -> bool:
       for bonus in _active_bonuses:
           if bonus.stat == stat and bonus.type == "flag":
               return true
       return false
   
   ## Get raw bonus value for a stat (for UI display).
   func get_bonus_value(stat: String) -> float:
       for bonus in _active_bonuses:
           if bonus.stat == stat:
               return bonus.value
       return 1.0  # Default multiplier
   
   ## Get all active bonuses (for debug/UI).
   func get_all_bonuses() -> Array[Dictionary]:
       return _active_bonuses.duplicate()
   ```

2. Add PassiveBonusManager to player scene (alongside AbilityManager)

## Task 2: Add Profession State to GameState

**Action:**
1. Modify `src/autoloads/game_state.gd` to add profession management:
   ```gdscript
   # Add to signals section (after existing signals)
   signal profession_selected(profession: ProfessionData)
   
   # Add to properties section
   var player_profession: ProfessionData = null
   
   # Add new method
   ## Select a profession for the player.
   ## Applies to player_instance if it exists, or defers until spawn.
   func select_profession(profession: ProfessionData) -> void:
       player_profession = profession
       profession_selected.emit(profession)
       
       # Apply to player if already spawned
       if player_instance and player_instance.has_method("set_profession"):
           player_instance.set_profession(profession)
           print("[GameState] Profession applied: %s" % profession.name)
       else:
           print("[GameState] Profession selected: %s (will apply on spawn)" % profession.name)
   ```
