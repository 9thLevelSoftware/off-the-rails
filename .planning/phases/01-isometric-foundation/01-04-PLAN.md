### Task 1: Isometric Level Scene

**Action**: Create `src/isometric/scenes/isometric_level.tscn`

Scene structure:
```
IsoLevel (Node2D)
├── IsoCanvas (isometric_canvas.gd)
│   ├── WorldLayer (Node2D, y_sort_enabled)
│   │   └── IsoTilemapAdapter (tilemap_adapter.gd)
│   │       └── TileMap (with iso_floor.tres)
│   ├── EntityLayer (Node2D, y_sort_enabled)
│   └── UILayer (CanvasLayer)
└── IsoCamera (camera_2d_controller.gd)
```

The scene should:
- Instantiate all adapter nodes
- Configure the TileMap with the placeholder tileset
- Set up proper parent-child relationships for Y-sorting
- Camera starts centered on origin

**Script**: Create `src/isometric/scenes/isometric_level.gd`

```gdscript
# Main isometric level scene controller
# Initializes and coordinates all isometric subsystems

class_name IsoLevel
extends Node2D

@onready var canvas: IsoCanvas = $IsoCanvas
@onready var camera: IsoCameraController = $IsoCamera
@onready var tilemap_adapter: IsoTilemapAdapter = $IsoCanvas/WorldLayer/IsoTilemapAdapter

func _ready() -> void:
    _initialize_level()

func _initialize_level() -> void:
    # Camera starts at origin
    camera.global_position = Vector2.ZERO
    
    # Generate a simple test floor (5x5 grid)
    _generate_test_floor()

func _generate_test_floor() -> void:
    if not tilemap_adapter or not tilemap_adapter.tilemap:
        push_warning("TileMap not ready")
        return
    
    # Fill a 5x5 area with floor tiles
    for x in range(-2, 3):
        for y in range(-2, 3):
            tilemap_adapter.set_tile(Vector2i(x, y), 0, Vector2i(0, 0))

# Set camera follow target
func set_camera_target(target: Node2D) -> void:
    camera.follow_target = target

# Get the tilemap adapter for external access
func get_tilemap() -> IsoTilemapAdapter:
    return tilemap_adapter
```

**Verify**: `ls src/isometric/scenes/isometric_level.tscn && ls src/isometric/scenes/isometric_level.gd`

**Done**: Main isometric level scene created with all adapters composed.

### Task 2: Test Scene

**Action**: Create `src/isometric/test/test_isometric_level.tscn`

Scene structure:
```
TestIsoLevel (Node2D)
├── IsoLevel (instance of isometric_level.tscn)
└── TestEntity (CharacterBody2D)
    ├── Sprite2D (placeholder 32x32 colored square)
    └── CollisionShape2D
```

**Script**: Create `src/isometric/test/test_isometric_level.gd`

```gdscript
# Test scene for verifying isometric foundation
# R1: TileMap + Y-sort verification
# R2: Camera follow + zoom verification

extends Node2D

@onready var iso_level: IsoLevel = $IsoLevel
@onready var test_entity: CharacterBody2D = $TestEntity

const MOVE_SPEED: float = 200.0

func _ready() -> void:
    # CRITIQUE FIX: Defer initialization to ensure child nodes are ready
    # This prevents race conditions with adapter _ready() calls
    call_deferred("_initialize_test")

func _initialize_test() -> void:
    # Validate required nodes
    if not iso_level:
        push_error("TestIsoLevel: IsoLevel child not found")
        return
    if not test_entity:
        push_error("TestIsoLevel: TestEntity child not found")
        return
    
    # Wait for tilemap adapter to be ready
    var tilemap_adapter = iso_level.get_tilemap()
    if tilemap_adapter and not tilemap_adapter.is_ready():
        push_warning("TileMap adapter not yet initialized, waiting...")
        await get_tree().process_frame
    
    # Connect camera to test entity
    iso_level.set_camera_target(test_entity)
    
    # Position test entity at origin
    test_entity.global_position = Vector2.ZERO
    
    print("Isometric Foundation Test")
    print("========================")
    print("WASD: Move test entity")
    print("Mouse wheel: Zoom in/out")
    print("Watch Y-sorting as entity moves")

func _physics_process(delta: float) -> void:
    _handle_movement(delta)

func _handle_movement(delta: float) -> void:
    var input_dir = Vector2.ZERO
    
    if Input.is_action_pressed("move_forward"):
        input_dir.y -= 1
    if Input.is_action_pressed("move_backward"):
        input_dir.y += 1
    if Input.is_action_pressed("move_left"):
        input_dir.x -= 1
    if Input.is_action_pressed("move_right"):
        input_dir.x += 1
    
    if input_dir != Vector2.ZERO:
        input_dir = input_dir.normalized()
        # CRITIQUE FIX: Proper isometric input conversion
        # For 2:1 isometric (64x32 tiles):
        # - Screen "up" should move entity visually upward (negative screen Y)
        # - Screen "right" should move entity visually rightward (positive screen X)
        # Godot's TileMap isometric uses diamond-down projection where:
        # - Grid +X goes diagonally down-right on screen
        # - Grid +Y goes diagonally down-left on screen
        # For intuitive WASD (W=up, A=left on SCREEN), we apply direct screen movement:
        var screen_dir = Vector2(input_dir.x, input_dir.y)
        test_entity.velocity = screen_dir * MOVE_SPEED
        test_entity.move_and_slide()
        # NOTE: This gives screen-relative movement. For grid-aligned movement,
        # convert using IsoViewportCalculator.screen_to_iso() for grid snapping.
```

**Verify**: `ls src/isometric/test/test_isometric_level.tscn && ls src/isometric/test/test_isometric_level.gd`

**Done**: Test scene created with movable entity for verification.
