Create `src/data/content_registry.gd` as the single boundary for all game content.

Structure:
```gdscript
class_name ContentRegistry
extends RefCounted

## Single source of truth for all game content (base + mods)
## Clean Architecture boundary — domain code queries this, never raw files

signal content_loaded(content_type: String, count: int)
signal content_merged(mod_id: String, content_type: String, count: int)
signal content_conflict(content_type: String, id: String, winner: String)

# Per-content-type registries (composition, not inheritance)
var items: ItemRegistry
var recipes: RecipeRegistry
var professions: ProfessionRegistry
var train_cars: TrainCarRegistry

var _data_loader: DataLoader
var _base_loaded: bool = false
var _loaded_mods: Array[String] = []

func _init() -> void:
    _data_loader = DataLoader.new()
    items = ItemRegistry.new()
    recipes = RecipeRegistry.new()
    professions = ProfessionRegistry.new()
    train_cars = TrainCarRegistry.new()

func load_base_content() -> bool:
    # Load all base game content from res://data/
    # Returns true if all content loaded successfully
    if _base_loaded:
        push_warning("Base content already loaded")
        return true
    
    var success := true
    success = success and _load_content_type("items", "res://data/base_items.json", items)
    success = success and _load_content_type("recipes", "res://data/base_recipes.json", recipes)
    # Add other content types as they're implemented
    
    _base_loaded = success
    return success

func merge_mod_content(mod_id: String, manifest: ModManifest) -> int:
    # Merge a mod's content into the registries
    # Returns count of content items merged
    # Emits content_conflict for ID collisions (mod wins)
    # CRITICAL: This MUST be fully implemented, not left as pass
    
    if mod_id in _loaded_mods:
        push_warning("Mod %s already loaded" % mod_id)
        return 0
    
    var total_count := 0
    var mod_base_path := "user://mods/%s/" % mod_id
    
    for content_file in manifest.content_files:
        var full_path := mod_base_path + content_file
        var data := _data_loader.load_json(full_path)
        if data.is_empty():
            push_warning("Failed to load mod content: %s" % full_path)
            continue
        
        # Determine content type from filename or data structure
        if data.has("items"):
            var count := _merge_items(mod_id, data.items)
            total_count += count
            content_merged.emit(mod_id, "items", count)
        if data.has("recipes"):
            var count := _merge_recipes(mod_id, data.recipes)
            total_count += count
            content_merged.emit(mod_id, "recipes", count)
        # Add other content types as needed
    
    _loaded_mods.append(mod_id)
    return total_count

func _merge_items(mod_id: String, items_data: Array) -> int:
    var count := 0
    for item_dict in items_data:
        var item := ItemRegistry.ItemData.from_dict(item_dict)
        if item:
            # Check for conflict
            if items.has_item(item.id):
                content_conflict.emit("items", item.id, mod_id)
            items.register_item(item)
            count += 1
    return count

func _merge_recipes(mod_id: String, recipes_data: Array) -> int:
    var count := 0
    for recipe_dict in recipes_data:
        var recipe := RecipeRegistry.RecipeData.from_dict(recipe_dict)
        if recipe:
            if recipes.has_recipe(recipe.id):
                content_conflict.emit("recipes", recipe.id, mod_id)
            recipes.register_recipe(recipe)
            count += 1
    return count

func get_item(id: String) -> ItemData:
    return items.get_item(id)

func get_recipe(id: String) -> RecipeData:
    return recipes.get_recipe(id)

func get_all_items() -> Array[ItemData]:
    return items.get_all()

func get_all_recipes() -> Array[RecipeData]:
    return recipes.get_all()

func _load_content_type(type_name: String, path: String, registry: RefCounted) -> bool:
    var data := _data_loader.load_json(path)
    if data.is_empty():
        push_error("Failed to load %s from %s" % [type_name, path])
        return false
    
    var count := registry.load_from_data(data)
    content_loaded.emit(type_name, count)
    return true
```

Design principles:
- This is the ONLY place domain code gets content from
- Registries are composed (not inherited) for clear ownership
- Signals enable UI/debugging to track content state
- Conflict resolution: mod content wins, emit signal for logging

Create `src/data/data_loader.gd` using JSON format (Godot-native, no parsing issues).

CRITICAL DECISION: Use JSON instead of YAML because:
- Godot 4.6 has native JSON.parse_string() support
- No external libraries required
- Existing design YAML docs will be converted to JSON at build time via a simple script
- Mods use JSON for maximum compatibility

Structure:
```gdscript
class_name DataLoader
extends RefCounted

## Infrastructure layer — converts JSON files to Godot Dictionaries
## Domain code never uses this directly; ContentRegistry mediates all access
## Uses JSON (Godot-native) — existing YAML design docs converted at build time

func load_json(path: String) -> Dictionary:
    # Load and parse a JSON file, return as Dictionary
    # Returns empty Dictionary on error (logged, not thrown)
    if not FileAccess.file_exists(path):
        push_error("JSON file not found: %s" % path)
        return {}
    
    var file := FileAccess.open(path, FileAccess.READ)
    if not file:
        push_error("Cannot open JSON file: %s (error: %s)" % [path, FileAccess.get_open_error()])
        return {}
    
    var content := file.get_as_text()
    file.close()
    
    var json := JSON.new()
    var error := json.parse(content)
    if error != OK:
        push_error("JSON parse error in %s at line %d: %s" % [path, json.get_error_line(), json.get_error_message()])
        return {}
    
    var data = json.get_data()
    if not data is Dictionary:
        push_error("JSON root must be object in %s" % path)
        return {}
    
    return data

func load_json_from_string(content: String) -> Dictionary:
    var json := JSON.new()
    var error := json.parse(content)
    if error != OK:
        push_error("JSON parse error at line %d: %s" % [json.get_error_line(), json.get_error_message()])
        return {}
    
    var data = json.get_data()
    return data if data is Dictionary else {}

func save_json(path: String, data: Dictionary, indent: int = 2) -> bool:
    # Save Dictionary as JSON (for mod tools, debugging)
    var content := JSON.stringify(data, "  " if indent > 0 else "")
    var file := FileAccess.open(path, FileAccess.WRITE)
    if not file:
        push_error("Cannot write JSON file: %s" % path)
        return false
    file.store_string(content)
    file.close()
    return true
```

Create per-content-type registries in `src/data/registries/`.

Each registry follows the same pattern:
1. Typed data class for the content type
2. Dictionary storage by ID
3. Load/register/get/getall methods
4. No file I/O — receives data from ContentRegistry

`src/data/registries/item_registry.gd`:
```gdscript
class_name ItemRegistry
extends RefCounted

## Registry for Item content type

var _items: Dictionary = {}  # id -> ItemData

func register_item(item: ItemData) -> void:
    _items[item.id] = item

func get_item(id: String) -> ItemData:
    return _items.get(id)

func get_all() -> Array[ItemData]:
    var result: Array[ItemData] = []
    for item in _items.values():
        result.append(item)
    return result

func has_item(id: String) -> bool:
    return id in _items

func get_count() -> int:
    return _items.size()

func load_from_data(data: Dictionary) -> int:
    # Parse items array from YAML data, register each
    if not data.has("items"):
        return 0
    
    var count := 0
    for item_data in data.items:
        var item := ItemData.from_dict(item_data)
        if item:
            register_item(item)
            count += 1
    return count

func clear() -> void:
    _items.clear()


class ItemData extends RefCounted:
    var id: String
    var name: String
    var description: String
    var category: String
    var stack_size: int = 1
    
    static func from_dict(data: Dictionary) -> ItemData:
        var item := ItemData.new()
        item.id = data.get("id", "")
        item.name = data.get("name", "")
        item.description = data.get("description", "")
        item.category = data.get("category", "misc")
        item.stack_size = data.get("stack_size", 1)
        return item if item.id else null
```
