Create `src/scripting/mod_api.gd` as the typed interface for mod scripts.

Structure:
```gdscript
class_name ModAPI
extends RefCounted

## Typed API for mod scripts to interact with game systems
## Mods call these methods to register content — no direct registry access
## Design: Mods can ADD content, not mutate or delete existing content

signal content_registered(mod_id: String, content_type: String, content_id: String)
signal script_executed(mod_id: String, script_path: String)

var _content_registry: ContentRegistry
var _current_mod_id: String = ""

func _init(content_registry: ContentRegistry) -> void:
    _content_registry = content_registry

func set_current_mod(mod_id: String) -> void:
    # Called by ModLoader before executing mod scripts
    _current_mod_id = mod_id

# ========== Content Registration ==========

func register_item(item_data: Dictionary) -> bool:
    # Register a new item from mod
    # item_data must have: id, name, description, category
    # Returns true on success
    if not _validate_item_data(item_data):
        push_error("[ModAPI] Invalid item data from mod '%s'" % _current_mod_id)
        return false
    
    # Prefix ID with mod_id to avoid collisions
    var prefixed_id := "%s_%s" % [_current_mod_id, item_data.id]
    item_data.id = prefixed_id
    
    var item := ItemRegistry.ItemData.from_dict(item_data)
    _content_registry.items.register_item(item)
    content_registered.emit(_current_mod_id, "item", prefixed_id)
    return true

func register_recipe(recipe_data: Dictionary) -> bool:
    # Register a new recipe from mod
    # recipe_data must have: id, name, station, inputs, output
    if not _validate_recipe_data(recipe_data):
        push_error("[ModAPI] Invalid recipe data from mod '%s'" % _current_mod_id)
        return false
    
    var prefixed_id := "%s_%s" % [_current_mod_id, recipe_data.id]
    recipe_data.id = prefixed_id
    
    var recipe := RecipeRegistry.RecipeData.from_dict(recipe_data)
    _content_registry.recipes.register_recipe(recipe)
    content_registered.emit(_current_mod_id, "recipe", prefixed_id)
    return true

func register_profession(profession_data: Dictionary) -> bool:
    # Register a new profession from mod
    pass

func register_train_car(car_data: Dictionary) -> bool:
    # Register a new train car from mod
    pass

# ========== Content Query (Read-Only) ==========

func get_item(id: String) -> Dictionary:
    # Return item data as Dictionary (safe copy, not reference)
    var item := _content_registry.get_item(id)
    return item.to_dict() if item else {}

func get_all_item_ids() -> Array[String]:
    # Return all registered item IDs
    var ids: Array[String] = []
    for item in _content_registry.get_all_items():
        ids.append(item.id)
    return ids

func item_exists(id: String) -> bool:
    return _content_registry.items.has_item(id)

# ========== Validation ==========

func _validate_item_data(data: Dictionary) -> bool:
    return data.has("id") and data.has("name")

func _validate_recipe_data(data: Dictionary) -> bool:
    return data.has("id") and data.has("station") and data.has("inputs") and data.has("output")
```

Design principles:
- Mods ONLY register content through ModAPI, never directly to registries
- ID prefixing prevents collisions between mods
- Read methods return copies (Dictionaries), not references
- Validation catches malformed content early
- Signals enable debugging/logging of mod activity

Create `src/scripting/event_hooks.gd` as the signal bus for mod scripts.

Structure:
```gdscript
class_name EventHooks
extends Node

## Signal bus for mod scripts to listen to game events
## Game systems emit signals here; mods connect handlers
## Singleton pattern — accessed via EventHooks autoload

# ========== Lifecycle Events ==========
signal game_ready()
signal game_paused()
signal game_resumed()
signal game_exiting()

# ========== Content Events ==========
signal item_registered(item_id: String, mod_id: String)
signal recipe_registered(recipe_id: String, mod_id: String)
signal content_loaded(content_type: String, count: int)

# ========== Train Events ==========
signal train_car_entered(car_id: String, player_id: int)
signal train_car_exited(car_id: String, player_id: int)
signal subsystem_state_changed(car_id: String, subsystem: String, new_state: int)

# ========== Expedition Events ==========
signal expedition_started(location_id: String)
signal expedition_ended(location_id: String, success: bool)
signal escalation_changed(old_level: int, new_level: int)

# ========== Crafting Events ==========
signal craft_started(recipe_id: String, station_id: String)
signal craft_completed(recipe_id: String, station_id: String, item_id: String)
signal craft_failed(recipe_id: String, reason: String)

# ========== Player Events ==========
signal player_joined(player_id: int, profession: String)
signal player_left(player_id: int)
signal player_inventory_changed(player_id: int, item_id: String, delta: int)

# ========== Mod Lifecycle ==========
signal mod_loading(mod_id: String)
signal mod_loaded(mod_id: String)
signal mod_unloaded(mod_id: String)

func _ready() -> void:
    # EventHooks is an autoload — available globally
    pass

# ========== Utility Methods ==========

func emit_custom(event_name: String, data: Dictionary = {}) -> void:
    # For game systems to emit custom events
    # Mods can listen via: EventHooks.connect(event_name, handler)
    # Note: This uses Godot's dynamic signal emission
    if has_signal(event_name):
        emit_signal(event_name, data)
    else:
        push_warning("[EventHooks] Unknown event: %s" % event_name)
```

Design principles:
- All game events mods might care about are defined here
- Mods connect to signals, game systems emit them
- Signals are typed where possible for autocomplete
- Custom events supported for extensibility
- No gameplay logic here — just signal definitions

Updates to ModLoader:
```gdscript
# Add to class variables
var _content_registry: ContentRegistry
var _mod_api: ModAPI
var _initialized: bool = false  # CRITICAL: initialization guard

func _ready() -> void:
    # CRITICAL: Use call_deferred to ensure other autoloads (EventHooks) are ready
    call_deferred("_initialize")

func _initialize() -> void:
    # Guard against multiple initialization
    if _initialized:
        push_warning("[ModLoader] Already initialized")
        return
    
    # CRITICAL: Verify EventHooks is available before proceeding
    if not is_instance_valid(EventHooks):
        push_error("[ModLoader] FATAL: EventHooks autoload not available. Check autoload order in project.godot.")
        return
    
    _error_handler = ModErrorHandler.new()
    add_child(_error_handler)
    
    # Initialize content registry and API
    _content_registry = ContentRegistry.new()
    _mod_api = ModAPI.new(_content_registry)
    
    # Load base game content first
    if not _content_registry.load_base_content():
        push_error("[ModLoader] Failed to load base game content")
        return
    
    # Discover and load mods
    var mods := discover_mods()
    print("[ModLoader] Discovered %d mods" % mods.size())
    
    var loaded := load_all_mods()
    print("[ModLoader] Loaded %d mods successfully" % loaded)
    
    _initialized = true
    
    # Emit ready signal
    all_mods_loaded.emit(loaded)
    EventHooks.game_ready.emit()

func is_ready() -> bool:
    # Guard method for other systems to check initialization state
    return _initialized

func load_mod(mod_id: String) -> bool:
    # Guard: ensure ModLoader is initialized
    if not _initialized:
        push_error("[ModLoader] Cannot load mod before initialization")
        return false
    
    var manifest := get_mod_manifest(mod_id)
    if not manifest:
        return false
    
    EventHooks.mod_loading.emit(mod_id)
    
    # Load content files via ContentRegistry
    var content_count := _content_registry.merge_mod_content(mod_id, manifest)
    
    # Execute mod scripts via ModAPI
    _mod_api.set_current_mod(mod_id)
    for script_path in manifest.scripts:
        _execute_mod_script(mod_id, script_path)
    
    _loaded_mods.append(mod_id)
    mod_loaded.emit(mod_id)
    EventHooks.mod_loaded.emit(mod_id)
    return true

func _execute_mod_script(mod_id: String, script_path: String) -> void:
    # Load and execute a mod's GDScript file
    # Script receives ModAPI as parameter for registration
    var full_path := "user://mods/%s/%s" % [mod_id, script_path]
    
    # CRITICAL: Validate file exists BEFORE attempting load
    if not FileAccess.file_exists(full_path):
        _error_handler.handle_error(mod_id, ModErrorHandler.ErrorType.SCRIPT_LOAD_FAILED, 
            "Script not found: %s" % script_path)
        return
    
    # CRITICAL: Validate this is actually a GDScript file
    if not full_path.ends_with(".gd"):
        _error_handler.handle_error(mod_id, ModErrorHandler.ErrorType.SCRIPT_LOAD_FAILED,
            "Invalid script extension (must be .gd): %s" % script_path)
        return
    
    # Attempt to load the script
    var script = load(full_path)
    if script == null:
        _error_handler.handle_error(mod_id, ModErrorHandler.ErrorType.SCRIPT_LOAD_FAILED,
            "Failed to load script (syntax error or load failure): %s" % script_path)
        return
    
    if not script is GDScript:
        _error_handler.handle_error(mod_id, ModErrorHandler.ErrorType.SCRIPT_LOAD_FAILED,
            "Loaded resource is not GDScript: %s" % script_path)
        return
    
    # Create instance and call init with error handling
    var instance = script.new()
    if instance == null:
        _error_handler.handle_error(mod_id, ModErrorHandler.ErrorType.SCRIPT_LOAD_FAILED,
            "Failed to instantiate script: %s" % script_path)
        return
    
    if instance.has_method("_mod_init"):
        # Wrap in try-catch pattern using call()
        instance.call("_mod_init", _mod_api)
    
    _mod_api.script_executed.emit(mod_id, script_path)
    print("[ModLoader] Executed mod script: %s/%s" % [mod_id, script_path])

# Add accessor for other systems
func get_content_registry() -> ContentRegistry:
    return _content_registry

func get_mod_api() -> ModAPI:
    return _mod_api
```
