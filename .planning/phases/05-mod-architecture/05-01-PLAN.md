```gdscript
class_name ModErrorHandler
extends Node

## Handles mod loading errors gracefully — logs, emits signals, never crashes

signal error_occurred(error: ModError)
signal warning_occurred(warning: ModWarning)

var _errors: Array[ModError] = []
var _warnings: Array[ModWarning] = []

func handle_error(mod_id: String, error_type: ErrorType, message: String, details: Dictionary = {}) -> ModError:
    # Create and log an error, emit signal, return error object
    # Never throws or crashes — always returns gracefully
    pass

func handle_warning(mod_id: String, warning_type: WarningType, message: String) -> ModWarning:
    # Create and log a warning, emit signal
    pass

func get_errors_for_mod(mod_id: String) -> Array[ModError]:
    # Return all errors for a specific mod
    pass

func get_all_errors() -> Array[ModError]:
    return _errors.duplicate()

func has_critical_errors() -> bool:
    # Returns true if any error would prevent game from running
    pass

func clear() -> void:
    _errors.clear()
    _warnings.clear()

enum ErrorType {
    MANIFEST_NOT_FOUND,
    MANIFEST_PARSE_FAILED,
    MANIFEST_INVALID,
    DEPENDENCY_MISSING,
    CONTENT_LOAD_FAILED,
    SCRIPT_LOAD_FAILED,
}

enum WarningType {
    DEPRECATED_FIELD,
    OPTIONAL_DEPENDENCY_MISSING,
    CONTENT_OVERRIDE,
}

class ModError extends RefCounted:
    var mod_id: String
    var error_type: ErrorType
    var message: String
    var details: Dictionary
    var timestamp: int
    
    func _to_string() -> String:
        return "[ModError] %s: %s - %s" % [mod_id, ErrorType.keys()[error_type], message]

class ModWarning extends RefCounted:
    var mod_id: String
    var warning_type: WarningType
    var message: String
    var timestamp: int
```

Design principles:
- NEVER crash or throw on error — always log and continue
- Return structured error objects for programmatic handling
- Emit signals so UI can display error state
- Distinguish errors (mod won't load) from warnings (mod loads with issues)
- Include timestamp for debugging load order issues
