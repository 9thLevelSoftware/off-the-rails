```gdscript
class_name InteractionController
extends Node

## Handles player interaction with train cars and other Interactables.
## Listens for E-key input and dispatches to the nearest valid target.

## The player node to track for interaction range
@export var player: CharacterBody3D

## Maximum distance for interaction
@export var interaction_range: float = 3.0

## Currently highlighted interactable (nearest valid target)
var current_target: Node = null

## Currently active interaction (if any)
var active_interaction: Node = null

## All interactables in range
var _interactables_in_range: Array[Area3D] = []

## Emitted when interaction target changes
signal target_changed(new_target: Node)

## Emitted when interaction occurs
signal interaction_occurred(target: Node, success: bool)


func _ready() -> void:
    set_process_input(true)
    
    # [CRITIQUE-FIX] Use call_deferred to handle player group timing
    # Player may not be in group yet if InteractionController._ready() runs first
    if not player:
        call_deferred("_deferred_find_player")


## [CRITIQUE-FIX] Deferred player lookup to handle initialization timing
func _deferred_find_player() -> void:
    await get_tree().process_frame
    _find_player()
    if not player:
        # Still no player - try again next frame
        await get_tree().process_frame
        _find_player()


func _find_player() -> void:
    # Try to get player from GameState
    var game_state = get_node_or_null("/root/GameState")
    if game_state and game_state.has_method("get") and game_state.get("player_instance"):
        player = game_state.player_instance
    else:
        # Fallback: find in scene
        var players = get_tree().get_nodes_in_group("player")
        if players.size() > 0:
            player = players[0]


func _input(event: InputEvent) -> void:
    if event.is_action_pressed("interact"):
        _on_interact_pressed()


func _on_interact_pressed() -> void:
    if active_interaction:
        # End current interaction
        _end_current_interaction()
        return
    
    if current_target:
        _try_interact_with(current_target)


func _try_interact_with(target: Node) -> void:
    # Check if target implements Interactable interface
    if target.has_method("interact"):
        var success = target.interact(player)
        interaction_occurred.emit(target, success)
        
        if success:
            active_interaction = target
            print("[Interaction] Started interaction with: %s" % target.name)
        else:
            print("[Interaction] Failed to interact with: %s" % target.name)
    else:
        push_warning("InteractionController: Target %s doesn't have interact method" % target.name)


func _end_current_interaction() -> void:
    if active_interaction and active_interaction.has_method("end_interaction"):
        active_interaction.end_interaction(player)
        print("[Interaction] Ended interaction with: %s" % active_interaction.name)
    active_interaction = null


func _process(_delta: float) -> void:
    _update_current_target()


func _update_current_target() -> void:
    if not player:
        return
    
    var nearest: Node = null
    var nearest_distance := interaction_range
    
    # Find all TrainCars with InteractionArea
    var train_cars = get_tree().get_nodes_in_group("train_car")
    for car in train_cars:
        if not car is TrainCar:
            continue
        
        var distance = player.global_position.distance_to(car.global_position)
        if distance < nearest_distance:
            nearest = car
            nearest_distance = distance
    
    # Update target if changed
    if nearest != current_target:
        current_target = nearest
        target_changed.emit(current_target)
        
        if current_target:
            print("[Interaction] Target: %s (%.1fm)" % [current_target.name, nearest_distance])


## Register an interactable area (called by areas entering detection zone)
func register_interactable(area: Area3D) -> void:
    if area not in _interactables_in_range:
        _interactables_in_range.append(area)


## Unregister an interactable area
func unregister_interactable(area: Area3D) -> void:
    _interactables_in_range.erase(area)


## Get current interaction prompt text
func get_current_prompt() -> String:
    if active_interaction:
        return "Press E to stop interacting"
    if current_target and current_target.has_method("get_interaction_prompt"):
        return current_target.get_interaction_prompt()
    if current_target:
        return "Press E to interact with %s" % current_target.name
    return ""
```

1. Create `src/train/train_manager.gd`:

```gdscript
class_name TrainManager
extends Node

## Central orchestrator for train cars and systems.
## Manages car instantiation, power flow, and inter-car dependencies.

## Factory for creating cars
var _factory: TrainCarFactory

## All cars currently on the train
var _cars: Array[TrainCar] = []

## Quick access to key cars
var engine_car: EngineCar
var workshop_car: WorkshopCar

## Node to parent car instances to
@export var cars_container: Node3D

## Emitted when a car is added to the train
signal car_added(car: TrainCar)

## Emitted when a car is removed from the train
signal car_removed(car: TrainCar)

## Emitted when train power state changes
signal train_power_changed(is_powered: bool)


func _ready() -> void:
    _factory = TrainCarFactory.new()
    
    # [CRITIQUE-FIX] Two-phase initialization:
    # Phase 1: Wait for scene tree to be ready
    await get_tree().process_frame
    
    # Phase 2: Create cars (separate from dependency wiring)
    _create_starting_cars()
    
    # Phase 3: Wire dependencies after all cars exist
    await get_tree().process_frame
    connect_car_dependencies()
    
    # Phase 4: Start systems
    _start_train()


## [CRITIQUE-FIX] Renamed from _initialize_starting_cars - only creates cars, no wiring
func _create_starting_cars() -> void:
    print("[TrainManager] Creating starting cars...")
    
    # Create engine car
    engine_car = _factory.create_car("engine") as EngineCar
    if engine_car:
        _add_car(engine_car, Vector3.ZERO)
        engine_car.add_to_group("train_car")
        print("[TrainManager] Engine car created at position 0")
    
    # Create workshop car
    workshop_car = _factory.create_car("workshop") as WorkshopCar
    if workshop_car:
        # Position workshop behind engine (Z+ is rear)
        _add_car(workshop_car, Vector3(0, 0, 10))
        workshop_car.add_to_group("train_car")
        print("[TrainManager] Workshop car created at position 1")
    
    # Connect dependencies
    connect_car_dependencies()
    
    # Start the train (bring systems online)
    _start_train()


func _add_car(car: TrainCar, position: Vector3) -> void:
    if cars_container:
        cars_container.add_child(car)
    else:
        add_child(car)
    
    car.global_position = position
    _cars.append(car)
    car_added.emit(car)


## Connect inter-car dependencies (Workshop needs Engine power)
func connect_car_dependencies() -> void:
    if workshop_car and engine_car:
        workshop_car.connect_to_engine(engine_car)
        print("[TrainManager] Workshop connected to Engine power grid")
    
    # Connect to engine power state for train-wide power signal
    if engine_car and engine_car.power_grid:
        engine_car.power_grid.power_availability_changed.connect(_on_power_changed)


func _on_power_changed(is_powered: bool) -> void:
    train_power_changed.emit(is_powered)
    print("[TrainManager] Train power: %s" % ("ON" if is_powered else "OFF"))


func _start_train() -> void:
    print("[TrainManager] Starting train systems...")
    
    # Start engine first (provides power)
    if engine_car:
        engine_car.start_engine()
        print("[TrainManager] Engine started - Power: %s" % engine_car.power_grid.is_providing_power())
    
    # Workshop comes online automatically when power is available
    # But we can explicitly activate it
    await get_tree().create_timer(0.1).timeout  # Brief delay for power signal
    
    if workshop_car:
        workshop_car.activate_workshop()
        print("[TrainManager] Workshop activated - Can craft: %s" % workshop_car.can_craft())


## Get all cars on the train
func get_cars() -> Array[TrainCar]:
    return _cars


## Get a car by ID
func get_car(car_id: String) -> TrainCar:
    for car in _cars:
        if car.car_id == car_id:
            return car
    return null


## Check if train has power
func has_power() -> bool:
    return engine_car != null and engine_car.power_grid != null and engine_car.power_grid.is_providing_power()


## Get train status summary for UI
func get_status_summary() -> Dictionary:
    var summary := {
        "has_power": has_power(),
        "car_count": _cars.size(),
        "cars": []
    }
    for car in _cars:
        summary["cars"].append(car.get_status_summary())
    return summary


## Shut down all train systems
func shutdown() -> void:
    print("[TrainManager] Shutting down train...")
    
    # Deactivate workshop first
    if workshop_car:
        workshop_car.deactivate_workshop()
    
    # Then stop engine
    if engine_car:
        engine_car.stop_engine()


## Debug: print train status
func print_status() -> void:
    print("=== Train Status ===")
    print("Power: %s" % ("ON" if has_power() else "OFF"))
    print("Cars: %d" % _cars.size())
    for car in _cars:
        var status = car.get_status_summary()
        print("  %s:" % status.car_name)
        for subsystem in status.subsystems:
            print("    - %s: %s" % [subsystem.name, subsystem.state])
    print("==================")
```

```gdscript
extends Node3D

## Train scene root - manages the train hub environment.

@onready var train_manager: TrainManager = $TrainManager
@onready var interaction_controller: InteractionController = $InteractionController

## Reference to cars container
@onready var cars_container: Node3D = $TrainCars


func _ready() -> void:
    # Register with GameState
    var game_state = get_node_or_null("/root/GameState")
    if game_state and game_state.has_method("register_scene"):
        game_state.register_scene(game_state.GameScene.TRAIN, self)
    
    # Setup train manager
    if train_manager:
        train_manager.cars_container = cars_container
        train_manager.train_power_changed.connect(_on_train_power_changed)
    
    # Debug: print status after initialization
    await get_tree().create_timer(0.5).timeout
    if train_manager:
        train_manager.print_status()


func _on_train_power_changed(is_powered: bool) -> void:
    print("[Train] Power state changed: %s" % ("ON" if is_powered else "OFF"))


## Called when player interacts with the train exit
func _on_exit_requested() -> void:
    var game_state = get_node_or_null("/root/GameState")
    if game_state and game_state.has_method("transition_to_expedition"):
        game_state.transition_to_expedition()
```
