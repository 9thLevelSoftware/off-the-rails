```gdscript
class_name Interactable
extends Node

## Interface for objects that can be interacted with via E-key.
## Implement this in any node that should respond to player interaction.

## Emitted when interaction starts
signal interaction_started(interactor: Node)

## Emitted when interaction ends
signal interaction_ended(interactor: Node)

## Display text shown when player can interact
@export var interaction_prompt: String = "Press E to interact"

## Whether interaction is currently available
@export var is_interactable: bool = true


## Override this method to handle the interaction
## Returns true if interaction was successful
func interact(interactor: Node) -> bool:
    if not is_interactable:
        return false
    
    interaction_started.emit(interactor)
    _on_interact(interactor)
    return true


## End the current interaction
func end_interaction(interactor: Node) -> void:
    interaction_ended.emit(interactor)
    _on_interaction_end(interactor)


## Override in subclasses to implement specific interaction behavior
func _on_interact(_interactor: Node) -> void:
    pass


## Override in subclasses to handle interaction end
func _on_interaction_end(_interactor: Node) -> void:
    pass


## Get the prompt text to display
func get_interaction_prompt() -> String:
    return interaction_prompt if is_interactable else ""
```

```gdscript
class_name TrainCar
extends Node3D

## Base class for all train cars using composition pattern.
## Holds subsystem instances and provides capability queries.

## Car identifier for factory and save/load
@export var car_id: String = ""

## Display name for UI
@export var car_name: String = "Train Car"

## Car's position in train (0 = engine, increases toward rear)
@export var car_position: int = 0

## Reference to the car's interaction area
@onready var interaction_area: Area3D = $InteractionArea

## Cached subsystems for quick access
var _subsystems: Array[Subsystem] = []

## Emitted when any subsystem state changes
signal subsystem_state_changed(subsystem: Subsystem, old_state: Subsystem.SubsystemState, new_state: Subsystem.SubsystemState)

## Emitted when car is interacted with
signal car_interacted(interactor: Node)


func _ready() -> void:
    _cache_subsystems()
    _connect_subsystem_signals()
    _setup_interaction_area()
    _on_car_ready()


## Cache all child subsystems for quick access
func _cache_subsystems() -> void:
    _subsystems.clear()
    for child in get_children():
        if child is Subsystem:
            _subsystems.append(child)


## Connect to all subsystem state change signals
func _connect_subsystem_signals() -> void:
    for subsystem in _subsystems:
        subsystem.state_changed.connect(_on_subsystem_state_changed.bind(subsystem))


## Setup interaction area signal
func _setup_interaction_area() -> void:
    if interaction_area:
        # Interaction area will be used by InteractionController
        pass


## Override in subclasses for car-specific initialization
func _on_car_ready() -> void:
    pass


## Handle subsystem state change
func _on_subsystem_state_changed(old_state: Subsystem.SubsystemState, new_state: Subsystem.SubsystemState, subsystem: Subsystem) -> void:
    subsystem_state_changed.emit(subsystem, old_state, new_state)


## Get all subsystems in this car
func get_subsystems() -> Array[Subsystem]:
    return _subsystems


## Get a subsystem by type
func get_subsystem(subsystem_type: GDScript) -> Subsystem:
    for subsystem in _subsystems:
        if subsystem.get_script() == subsystem_type:
            return subsystem
    return null


## Get subsystem by class name string
func get_subsystem_by_name(class_name_str: String) -> Subsystem:
    for subsystem in _subsystems:
        if subsystem.get_class() == class_name_str or subsystem.display_name == class_name_str:
            return subsystem
    return null


## Check if car has a specific capability (subsystem type that is operational)
func has_capability(subsystem_type: GDScript) -> bool:
    var subsystem = get_subsystem(subsystem_type)
    return subsystem != null and subsystem.is_operational()


## Bring all subsystems online (respecting dependencies)
func bring_all_online() -> void:
    for subsystem in _subsystems:
        subsystem.bring_online()


## Take all subsystems offline
func take_all_offline() -> void:
    for subsystem in _subsystems:
        subsystem.take_offline()


## Get car status summary for UI
func get_status_summary() -> Dictionary:
    var summary := {
        "car_name": car_name,
        "car_id": car_id,
        "subsystems": []
    }
    for subsystem in _subsystems:
        summary["subsystems"].append({
            "name": subsystem.display_name,
            "state": subsystem.get_state_string(),
            "operational": subsystem.is_operational()
        })
    return summary
```

```gdscript
class_name EngineCar
extends TrainCar

## Engine car - provides locomotion and power to the train.

## Quick access to subsystems
@onready var power_grid: PowerGrid = $PowerGrid
@onready var locomotion: Locomotion = $Locomotion


func _on_car_ready() -> void:
    car_id = "engine"
    car_name = "Engine"
    car_position = 0
    
    # Locomotion depends on Power Grid
    if locomotion and power_grid:
        power_grid.power_availability_changed.connect(_on_power_changed)


func _on_power_changed(power_available: bool) -> void:
    if power_available:
        locomotion.bring_online()
    else:
        locomotion.take_offline()


## Check if train can move
func can_move() -> bool:
    return locomotion != null and locomotion.is_operational()


## Get current speed multiplier
func get_speed_multiplier() -> float:
    return locomotion.get_speed_multiplier() if locomotion else 0.0


## Bring engine online (starts power grid first)
func start_engine() -> void:
    if power_grid:
        power_grid.bring_online()
    # Locomotion will come online automatically via signal


## Shut down engine
func stop_engine() -> void:
    if locomotion:
        locomotion.take_offline()
    if power_grid:
        power_grid.take_offline()
```

```gdscript
class_name TrainCarFactory
extends RefCounted

## Factory for creating train cars from data resources.
## Supports data-driven car instantiation for easy configuration.

## Map of car_id to scene path
const CAR_SCENES := {
    "engine": "res://src/train/cars/engine.tscn",
    "workshop": "res://src/train/cars/workshop.tscn"
}

## Cached loaded scenes
var _scene_cache: Dictionary = {}


## Create a car instance by ID
func create_car(car_id: String) -> TrainCar:
    var scene_path = CAR_SCENES.get(car_id)
    if not scene_path:
        push_error("TrainCarFactory: Unknown car_id '%s'" % car_id)
        return null
    
    var scene = _get_or_load_scene(scene_path)
    if not scene:
        push_error("TrainCarFactory: Failed to load scene '%s'" % scene_path)
        return null
    
    var car_instance = scene.instantiate() as TrainCar
    if not car_instance:
        push_error("TrainCarFactory: Scene '%s' root is not a TrainCar" % scene_path)
        return null
    
    return car_instance


## Create a car from a data resource
func create_car_from_data(car_data: Resource) -> TrainCar:
    if not car_data or not car_data.has_method("get_car_id"):
        push_error("TrainCarFactory: Invalid car data resource")
        return null
    
    var car = create_car(car_data.get_car_id())
    if car and car_data.has_method("apply_to_car"):
        car_data.apply_to_car(car)
    return car


## Create all starting cars for a new game
func create_starting_cars() -> Array[TrainCar]:
    var cars: Array[TrainCar] = []
    
    # Starting cars: Engine (damaged state), Workshop
    var engine = create_car("engine")
    if engine:
        cars.append(engine)
    
    var workshop = create_car("workshop")
    if workshop:
        cars.append(workshop)
    
    return cars


## Get or load a scene from cache
func _get_or_load_scene(path: String) -> PackedScene:
    if _scene_cache.has(path):
        return _scene_cache[path]
    
    var scene = load(path) as PackedScene
    if scene:
        _scene_cache[path] = scene
    return scene


## Clear the scene cache
func clear_cache() -> void:
    _scene_cache.clear()


## Get list of available car IDs
static func get_available_car_ids() -> Array[String]:
    var ids: Array[String] = []
    for key in CAR_SCENES.keys():
        ids.append(key)
    return ids
```
