## Crafting Lifecycle
1. Player selects recipe → RecipeValidator checks
2. Resources consumed → InventoryRepository.consume()
3. Job added to queue → CraftingEventBus.job_queued
4. Scheduler ticks time → Job progresses
5. Job completes → CraftingEventBus.job_completed
6. Output added → InventoryRepository.add()

## Task 4: ExpeditionPauseHandler (CRITIQUE FIX: Correct Signal Names)

**Action**: Create `src/crafting/infrastructure/expedition_pause_handler.gd` to pause crafting during expeditions.

**CRITIQUE FIX**: GameState uses `scene_transition_completed(new_scene: GameScene)` signal with enum, not `scene_changed(scene_name: String)`.

```gdscript
class_name ExpeditionPauseHandler
extends RefCounted

## Listens to game state changes and pauses crafting when in expedition.
## CRITIQUE FIX: Uses correct signal name and GameScene enum.

var scheduler: JobScheduler
var _connected: bool = false

func _init(scheduler: JobScheduler) -> void:
    self.scheduler = scheduler

func connect_to_game_state(game_state: Node) -> void:
    if _connected:
        return
    # CRITIQUE FIX: Use actual signal name from GameState
    if game_state.has_signal("scene_transition_completed"):
        game_state.scene_transition_completed.connect(_on_scene_transition_completed)
        _connected = true
    else:
        push_warning("ExpeditionPauseHandler: GameState missing scene_transition_completed signal")

func _on_scene_transition_completed(new_scene) -> void:
    # GameState.GameScene enum: TRAIN = 0, EXPEDITION = 1
    # Use duck typing to handle both enum and int
    var scene_value: int = new_scene if new_scene is int else int(new_scene)
    match scene_value:
        1:  # GameScene.EXPEDITION
            scheduler.pause("Expedition in progress")
        0:  # GameScene.TRAIN
            scheduler.resume()
        _:
            pass  # Unknown scene, maintain current state
```

## Task 5: GameState Inventory Extension (CRITIQUE FIX: Critical Blocker)

**CRITIQUE FIX**: This is a CRITICAL BLOCKER. GameState currently has NO inventory system. The entire crafting infrastructure depends on this.

Add to GameState (after existing properties around line 54):
```gdscript
# --- Inventory State (Phase 6: Crafting) ---
## Player inventory - item_id -> quantity
## Tracks both raw resources (scrap_metal, chemicals) and crafted items (bandage, stim_shot)
var inventory: Dictionary = {}

## Emitted when inventory changes
signal inventory_changed(item_id: String, old_quantity: int, new_quantity: int)

## Adds quantity of an item to inventory. Handles both resources and crafted items.
func add_to_inventory(item_id: String, quantity: int) -> void:
    var old_qty: int = inventory.get(item_id, 0)
    inventory[item_id] = old_qty + quantity
    inventory_changed.emit(item_id, old_qty, inventory[item_id])

## Removes quantity of an item from inventory. Returns false if insufficient.
func remove_from_inventory(item_id: String, quantity: int) -> bool:
    var current: int = inventory.get(item_id, 0)
    if current < quantity:
        return false
    var old_qty: int = current
    inventory[item_id] = current - quantity
    inventory_changed.emit(item_id, old_qty, inventory[item_id])
    return true

## Gets current quantity of an item.
func get_inventory_quantity(item_id: String) -> int:
    return inventory.get(item_id, 0)

## Checks if player has at least the specified quantity.
func has_inventory_quantity(item_id: String, quantity: int) -> bool:
    return get_inventory_quantity(item_id) >= quantity

## Checks if player has all items in the dictionary {item_id: quantity}.
func has_all_inventory(items: Dictionary) -> bool:
    for item_id in items:
        if not has_inventory_quantity(item_id, items[item_id]):
            return false
    return true

## Consumes all items in the dictionary. Returns false and consumes nothing if any missing.
func consume_inventory(items: Dictionary) -> bool:
    if not has_all_inventory(items):
        return false
    for item_id in items:
        remove_from_inventory(item_id, items[item_id])
    return true

## Adds all items in the dictionary.
func add_all_inventory(items: Dictionary) -> void:
    for item_id in items:
        add_to_inventory(item_id, items[item_id])

## Debug: Sets inventory for testing purposes.
func debug_set_inventory(items: Dictionary) -> void:
    inventory = items.duplicate()
    for item_id in inventory:
        inventory_changed.emit(item_id, 0, inventory[item_id])
```

**Note**: Scene tracking already exists via `scene_transition_completed(new_scene: GameScene)` signal.
