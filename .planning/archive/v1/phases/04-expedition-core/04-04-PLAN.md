```gdscript
func _ready() -> void:
    call_deferred("_deferred_connect_signals")
    call_deferred("_validate_spawn_points")

func _deferred_connect_signals() -> void:
    if not escalation_manager:
        # Try to find it
        var expedition = get_tree().get_first_node_in_group("expedition")
        if expedition:
            escalation_manager = expedition.get_node_or_null("EscalationManager")
    
    if escalation_manager:
        if not escalation_manager.threshold_crossed.is_connected(_on_threshold_crossed):
            escalation_manager.threshold_crossed.connect(_on_threshold_crossed)
    else:
        push_error("EnemySpawner: No EscalationManager found, spawning disabled")
```

<task id="5">
<action>
Implement threshold-based spawning with debounce:
```gdscript
var _last_processed_threshold: EscalationManager.EscalationThreshold = EscalationManager.EscalationThreshold.NORMAL

func _on_threshold_crossed(old_threshold: EscalationManager.EscalationThreshold, new_threshold: EscalationManager.EscalationThreshold) -> void:
    # CRITIQUE FIX: Debounce - only process if we haven't seen this threshold
    if new_threshold == _last_processed_threshold:
        return
    _last_processed_threshold = new_threshold
    
    var spawn_count := 0
    match new_threshold:
        EscalationManager.EscalationThreshold.ELEVATED:
            spawn_count = 1
        EscalationManager.EscalationThreshold.HIGH:
            spawn_count = 2
        EscalationManager.EscalationThreshold.CRITICAL:
            spawn_count = 3
        EscalationManager.EscalationThreshold.OVERRUN:
            spawn_count = 2
    
    if spawn_count > 0:
        var actual = _spawn_enemies(spawn_count)
        print("Spawned %d enemies at threshold %s" % [actual, EscalationManager.EscalationThreshold.keys()[new_threshold]])
```
