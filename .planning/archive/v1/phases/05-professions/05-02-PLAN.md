## Task 1: Create AbilityManager Component

**Action:**
1. Create `src/professions/ability_manager.gd`:
   ```gdscript
   class_name AbilityManager
   extends Node
   
   ## Manages profession abilities: cooldowns, input, and activation.
   ## Uses typed AbilityData from Plan 05-01 for parsed cooldown values.
   
   signal ability_activated(ability: AbilityData)
   signal ability_ready(ability_id: String)
   signal cooldown_started(ability_id: String, duration: float)
   
   var _profession: ProfessionData = null
   var _abilities: Array[AbilityData] = []  # Typed ability data
   var _cooldowns: Dictionary = {}  # ability_id -> remaining seconds
   var _effects_handler: AbilityEffects = null
   
   func _ready() -> void:
       # Create effects handler as child (ensures proper lifecycle)
       _effects_handler = AbilityEffects.new()
       _effects_handler.name = "AbilityEffects"
       add_child(_effects_handler)
   
   func set_profession(profession: ProfessionData) -> void:
       _profession = profession
       _abilities.clear()
       _cooldowns.clear()
       
       if profession == null:
           return
       
       # Convert dictionaries to typed AbilityData
       _abilities = AbilityData.from_profession(profession)
       for ability in _abilities:
           _cooldowns[ability.id] = 0.0
       
       print("[AbilityManager] Loaded %d abilities for %s" % [_abilities.size(), profession.name])
   
   func _process(delta: float) -> void:
       _update_cooldowns(delta)
   
   func _update_cooldowns(delta: float) -> void:
       for ability_id in _cooldowns:
           if _cooldowns[ability_id] > 0:
               _cooldowns[ability_id] = maxf(0.0, _cooldowns[ability_id] - delta)
               if _cooldowns[ability_id] == 0.0:
                   ability_ready.emit(ability_id)
   
   func _unhandled_input(event: InputEvent) -> void:
       if event.is_action_pressed("ability_1"):
           _try_activate_slot(0)
           get_viewport().set_input_as_handled()
       elif event.is_action_pressed("ability_2"):
           _try_activate_slot(1)
           get_viewport().set_input_as_handled()
       elif event.is_action_pressed("ability_3"):
           _try_activate_slot(2)
           get_viewport().set_input_as_handled()
   
   func _try_activate_slot(slot_index: int) -> bool:
       if slot_index < 0 or slot_index >= _abilities.size():
           return false
       
       var ability := _abilities[slot_index]
       
       if _cooldowns[ability.id] > 0:
           print("[AbilityManager] %s on cooldown (%.1fs remaining)" % [ability.display_name, _cooldowns[ability.id]])
           return false
       
       # Start cooldown
       _cooldowns[ability.id] = ability.cooldown_seconds
       cooldown_started.emit(ability.id, ability.cooldown_seconds)
       
       # Emit activation and execute effect
       ability_activated.emit(ability)
       _effects_handler.execute_ability(ability, get_parent())
       
       return true
   
   ## Get remaining cooldown for an ability (for UI)
   func get_cooldown_remaining(ability_id: String) -> float:
       return _cooldowns.get(ability_id, 0.0)
   
   ## Get cooldown progress 0.0 (ready) to 1.0 (just activated)
   func get_cooldown_progress(ability_id: String) -> float:
       for ability in _abilities:
           if ability.id == ability_id:
               if ability.cooldown_seconds <= 0:
                   return 0.0
               return _cooldowns.get(ability_id, 0.0) / ability.cooldown_seconds
       return 0.0
   
   ## Get ability at slot index (for UI)
   func get_ability_at_slot(slot_index: int) -> AbilityData:
       if slot_index < 0 or slot_index >= _abilities.size():
           return null
       return _abilities[slot_index]
   
   ## Get total ability count
   func get_ability_count() -> int:
       return _abilities.size()
   ```

## Task 2: Create Ability Effects Handler

**Action:**
1. Create `src/professions/ability_effects.gd`:
   ```gdscript
   class_name AbilityEffects
   extends Node
   
   ## Handles ability effect execution.
   ## Emits signals that game systems can subscribe to for actual effects.
   ## V1: Stub implementations that log and emit signals.
   
   # Engineer ability signals
   signal emergency_repair_activated(caster: Node)
   signal power_reroute_activated(caster: Node)
   signal system_overclock_activated(caster: Node)
   
   # Medic ability signals
   signal field_surgery_activated(caster: Node)
   signal stabilize_activated(caster: Node)
   signal purge_activated(caster: Node)
   
   ## Execute an ability's effect.
   ## Called by AbilityManager when ability activates.
   func execute_ability(ability: AbilityData, caster: Node) -> void:
       match ability.id:
           # Engineer abilities
           "emergency_repair":
               _do_emergency_repair(caster)
           "power_reroute":
               _do_power_reroute(caster)
           "system_overclock":
               _do_system_overclock(caster)
           # Medic abilities
           "field_surgery":
               _do_field_surgery(caster)
           "stabilize":
               _do_stabilize(caster)
           "purge":
               _do_purge(caster)
           _:
               push_warning("AbilityEffects: Unknown ability '%s'" % ability.id)
   
   # --- Engineer Abilities ---
   
   func _do_emergency_repair(caster: Node) -> void:
       print("[Ability] Emergency Repair activated!")
       # V1: Emit signal for train subsystem to handle
       # Future: Find nearest damaged subsystem, apply temporary repair
       emergency_repair_activated.emit(caster)
   
   func _do_power_reroute(caster: Node) -> void:
       print("[Ability] Power Reroute activated!")
       # V1: Emit signal for power system to handle
       # Future: Show UI to select power source/destination
       power_reroute_activated.emit(caster)
   
   func _do_system_overclock(caster: Node) -> void:
       print("[Ability] System Overclock activated!")
       # V1: Emit signal for subsystem boost
       # Future: Boost target subsystem output with damage risk
       system_overclock_activated.emit(caster)
   
   # --- Medic Abilities ---
   
   func _do_field_surgery(caster: Node) -> void:
       print("[Ability] Field Surgery activated!")
       # V1: Emit signal for healing system
       # Future: Complex healing outside infirmary
       field_surgery_activated.emit(caster)
   
   func _do_stabilize(caster: Node) -> void:
       print("[Ability] Stabilize activated!")
       # V1: Emit signal for downed player handling
       # Future: Prevent bleedout timer on downed player
       stabilize_activated.emit(caster)
   
   func _do_purge(caster: Node) -> void:
       print("[Ability] Purge activated!")
       # V1: Emit signal for status effect removal
       # Future: Clear poison/infection/radiation from target
       purge_activated.emit(caster)
   ```

- Each handler logs and emits a signal
- No hardcoded dependencies on systems that don't exist yet
